# Example GraphQL Schema

# This file is an example of a user-created schema.

# You can edit this schema file.


schema {
  query: Query
}

type Query {
  getBook(id: ID!): Book!
  getPeople(filter: String!): [Person!]!
}

input Cursor {
  start: Int!
  limit: Int!
}

type Book {
  id: ID!
  language: Language!
  author: Author!
  title: String!
  description: String!
  categories: [Category!]!
  translation(language: Language!): Book
}

# implemented by Author & User
interface Person {
  id: ID!
  name: String
}

type Author implements Person {
  id: ID!
  name: String!
  books: [Book!]!
}

type User implements Person {
  id: ID!
  name: String
  username: String!
}
